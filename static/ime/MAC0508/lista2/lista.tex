Lista de bigramas:

    \item $\text{\textless ini\textgreater}\mid \text{eu} = 1$
    \item $\text{\textless ini\textgreater}\mid \text{Mané} = 3$
    \item $\text{\textless ini\textgreater}\mid \text{bem} = 1$

    \item $\text{Mane} \mid \text{\textless fim\textgreater} = 2$

    \item $\text{sei}\mid \text{\textless fim\textgreater} = 2$
\end{itemize}

\subsection*{1. Qual a proxima palavra mais provavel prevista pelo modelo}
\begin{itemize}
    \item a. Mané (eu $P(eu|Mane) = 0.6$)
    \item b. Mané eu (sou|sei $P(sou|eu) = 0.4, P(sei|eu) = 0.4$)
    \item c. Mane eu sou (Mane|<fim> $P(Mane|sou) = 0.5, P(<fim>|sou) = 0.5$)
    \item d. bem eu sei (<fim> $P(<fim>|sei) = 0.66$) 
\end{itemize}
\subsection*{2. Qual das seguintes frases é melhor, ou seja, tem maior probabilidade com este modelo?}
\begin{itemize}
    \item a. $P(\text{Mane} \mid \text{\textless ini \textgreater})*P(\text{eu} \mid \text{Mane})*P(\text{bem} \mid \text{eu})*P(\text{eu} \mid \text{bem})*P(\text{sei} \mid \text{eu})*P(\text{\textless fim \textgreater} \mid \text{sei}) = 0.6*0.6*0.2*0.5*0.4*0.4 = 0.00576$
    \item b. $P(\text{Mane} \mid \text{\textless ini \textgreater})*P(\text{eu} \mid \text{Mane})*P(\text{sou} \mid \text{eu})*P(\text{\textless fim \textgreater} \mid \text{sou}) = 0.6*0.6*0*\dots = 0$
    \item c. $P(\text{eu} \mid \text{\textless ini \textgreater})*P(\text{bem} \mid \text{eu})*P(\text{sei} \mid \text{bem})*P(\text{Mane} \mid \text{sei})*P(\text{eu} \mid \text{Mane})*P(\text{sou} \mid \text{eu})*P(\text{\textless fim \textgreater} \mid \text{sou}) = 0.2*0.2*0.5*0.33*0.6*0.4*0.2 = 0.0003168$
\end{itemize}
Logo a frase \textbf{a} tem a melhor probabilidade
